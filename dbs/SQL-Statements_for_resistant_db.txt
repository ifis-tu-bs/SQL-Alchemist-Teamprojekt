CREATE TABLE Task(
    id int(10) unsigned auto_increment PRIMARY KEY,
    name varchar(200) NOT NULL,
    db_name varchar(200) NOT NULL,
    db_mem boolean,
    players int(10)
),

CREATE TABLE PLAYER(
    id int(10) unsigned PRIMARY KEY,
    task_id int(10) NOT NULL,
    FOREIGN KEY (task_id) REFERENCES TASK (id)
)
